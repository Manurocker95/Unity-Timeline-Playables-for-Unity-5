// The representation of the Timeline Clip
[System.Serializable]
public class #SCRIPTNAME# : PlayableAsset, ITimelineClipAsset
{
	public #SCRIPTNAME#Playable template = new #SCRIPTNAME#Playable();

	// GameObjects from Scenes must use ExposedReferences
	public ExposedReference<GameObject> exposedReference;
	
	// Create the runtime version of the clip, by creating a copy of the template
	public override PlayableHandle CreatePlayable(PlayableGraph graph, GameObject go)
    {
		var handle = graph.CloneScriptPlayable(template);
		handle.GetObject<#SCRIPTNAME#Playable>().sceneReference = exposedReference.Resolve(graph.resolver);
		return handle;
    }
	
	// Use this to tell the Timeline Editor what features this clip supports
	public ClipCaps clipCaps 
	{
		get { return ClipCaps.None; }
	}
}
